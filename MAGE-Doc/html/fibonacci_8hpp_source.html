23  inline void FibonacciSpiralSamplesInUnitCircle(XMFLOAT2 *samples, size_t nb_samples,
24  float shift = 1.0f, float alpha = 2.0f) {
25 
26  // Boundary points
27  // alpha == 0.0f -> jagged boundary.
28  // alpha == 2.0f -> smooth boundary.
29  const uint32_t np_boundary = static_cast< uint32_t >(alpha * sqrt(static_cast< float >(nb_samples))) + 1;
30 
31  size_t j = 0;
32  for (size_t i = j; i <= nb_samples - np_boundary; ++i, ++j, ++samples) {
33  const float r = sqrt((i + 0.5f) / (nb_samples - 0.5f * np_boundary));
34  const float phi = XM_GA * (i + shift);
35  *samples = XMFLOAT2(r * cos(phi), r * sin(phi));
36  }
37  for (size_t i = j + 1; i < nb_samples; ++i, ++samples) {
38  const float r = 1.0f;
39  const float phi = XM_GA * (i + shift);
40  *samples = XMFLOAT2(r * cos(phi), r * sin(phi));
41  }
42  }

44  inline void FibonacciSpiralSamplesOnUnitSphere(XMFLOAT3 *samples, size_t nb_samples,
45  float shift = 1.0f) {
46 
47  const float offset = 2.0f / nb_samples;
48

52  const float sin_theta = sqrt(1.0f - cos_theta * cos_theta);
