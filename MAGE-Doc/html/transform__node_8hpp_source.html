1801  inline void TransformNode::UpdateObjectToWorldMatrix() const noexcept {
1802  if (m_dirty_object_to_world) {
1803  if (HasParentNode()) {
1804  m_object_to_world = GetObjectToParentMatrix() * m_parent->GetTransform()->GetObjectToWorldMatrix();
1805  }
1806  else {
1807  m_object_to_world = GetObjectToParentMatrix();
1808  }
1809 
1810  m_dirty_object_to_world = false;
1811  }
1812  }

1814  inline void TransformNode::UpdateWorldToObjectMatrix() const noexcept {
1815  if (m_dirty_world_to_object) {
1816  if (HasParentNode()) {
1817  m_world_to_object = m_parent->GetTransform()->GetWorldToObjectMatrix() * GetParentToObjectMatrix();
1818  }
1819  else {
1820  m_world_to_object = GetParentToObjectMatrix();
1821  }
1822 
1823  m_dirty_world_to_object = false;
