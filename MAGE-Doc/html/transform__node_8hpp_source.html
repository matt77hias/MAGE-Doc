1526  inline void TransformNode::UpdateObjectToWorldMatrix() const {
1527  if (m_dirty_object_to_world) {
1528  if (HasParentNode()) {
1529  m_object_to_world = GetObjectToParentMatrix() * m_parent->GetTransform()->GetObjectToWorldMatrix();
1530  }
1531  else {
1532  m_object_to_world = GetObjectToParentMatrix();
1533  }
1534 
1535  m_dirty_object_to_world = false;
1536  }
1537  }

1543  }

1545  m_world_to_object = GetParentToObjectMatrix();
1546  }
1547
