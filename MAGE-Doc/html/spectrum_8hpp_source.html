29  inline const XMVECTOR XM_CALLCONV LinearToSRGB(FXMVECTOR linear) noexcept {
30  // Frostbite's conversion
31  static const float exp = 1.0f / 2.4f;
32 
33  const XMVECTOR low = linear * 12.92f;
34  const XMVECTOR high = 1.055f * XMVectorPow(linear, XMVectorReplicate(exp))
35  - XMVectorReplicate(0.055f);
36  const XMVECTOR comp = XMVectorLessOrEqual(linear, XMVectorReplicate(0.0031308f));
37 
38  return XMVectorSetW(XMVectorSelect(high, low, comp), XMVectorGetW(linear));
39  }

49  inline const XMVECTOR XM_CALLCONV SRGBToLinear(FXMVECTOR srgb) noexcept {
50  // Frostbite's conversion
51  static const float mlow = 1.0f / 12.92f;
52  static const float mhigh = 1.0f / 1.055f;
53 
54  const XMVECTOR low = srgb * mlow;
55  const XMVECTOR high = XMVectorPow(
56  mhigh * (srgb + XMVectorReplicate(0.055f)),
57  XMVectorReplicate(2.4f));
58  const XMVECTOR comp = XMVectorLessOrEqual(srgb, XMVectorReplicate(0.04045f));
59 
60  return XMVectorSetW(XMVectorSelect(high, low, comp), XMVectorGetW(srgb));
61  }
