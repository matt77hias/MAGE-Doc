376  const XMMATRIX rotation_m = XMMatrixRotationNormal(normal, angle);
377  XMFLOAT4X4 rotation;
378  XMStoreFloat4x4(&rotation, rotation_m);
379 
380  // cosf function instead of sinf in case the angles are greater than [-1,1]
381  m_rotation.y = -asinf(rotation._32);
382  const float cp = cosf(m_rotation.y);
383  const float cr = rotation._22 / cp;
384  m_rotation.z = acosf(cr);
385  const float cy = rotation._33 / cp;
386  m_rotation.x = acosf(cy);
387 
388  Update();
389  }
